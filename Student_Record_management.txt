public class StudentController {

    public Student__c stu {get; set;}

    public List<Student__c> studentList {get; set;}

    // CONSTRUCTOR
    public StudentController() {

        stu = new Student__c();

        loadStudents();
    }

    // LOAD STUDENT LIST
    public void loadStudents() {

        studentList = [
            SELECT Id,
                   Name,
                   RollNo__c,
                   class__c,
                   MobNo__c
            FROM Student__c
            ORDER BY CreatedDate DESC
        ];
    }

    // SAVE STUDENT
    public PageReference saveStudent() {

        insert stu;

        ApexPages.addMessage(
            new ApexPages.Message(
                ApexPages.Severity.CONFIRM,
                'Student Saved Successfully'
            )
        );

        stu = new Student__c();

        loadStudents();

        return null;
    }

    // EDIT STUDENT
    public PageReference editStudent() {

        String studentId =
            ApexPages.currentPage()
            .getParameters()
            .get('sid');

        stu = [
            SELECT Id,
                   Name,
                   RollNo__c,
                   class__c,
                   MobNo__c
            FROM Student__c
            WHERE Id = :studentId
        ];

        return null;
    }

    // UPDATE STUDENT
    public PageReference updateStudent() {

        update stu;

        ApexPages.addMessage(
            new ApexPages.Message(
                ApexPages.Severity.CONFIRM,
                'Student Updated Successfully'
            )
        );

        stu = new Student__c();

        loadStudents();

        return null;
    }

    // DELETE STUDENT
    public PageReference deleteStudent() {

        String studentId =
            ApexPages.currentPage()
            .getParameters()
            .get('sid');

        Student__c s = [
            SELECT Id
            FROM Student__c
            WHERE Id = :studentId
        ];

        delete s;

        loadStudents();

        ApexPages.addMessage(
            new ApexPages.Message(
                ApexPages.Severity.CONFIRM,
                'Student Deleted Successfully'
            )
        );

        return null;
    }
}
